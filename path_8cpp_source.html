143 void Path::sanitize()
144 {
145  // Replace any backslashes \ with forward slashes /.
146  size_t cur = string::npos;
147  while ((cur = m_path.find("\\")) != string::npos) { // assignment intended
148  m_path.replace(cur, 1, "/");
149  }
150 
151  // Replace all double slashes // with a single one
152  cur = string::npos;
153  while ((cur = m_path.find("//")) != string::npos) { // assignment intended
154  m_path.replace(cur, 2, "/");
155  }
156 
157  // Remove trailing slash if any (except for the filesystem root)
158  long len = m_path.length();
159 #if defined(_PATHIE_UNIX)
160  if (len > 1 && m_path[len - 1] == '/')
161  m_path = m_path.substr(0, len - 1);
162 #elif defined(_WIN32)
163  if (len > 1) { // / is root of current drive, "x" is the relative path "./x"
164  // Check if X:/foo/bar
165  if (len > 3 && m_path[len - 1] == '/') { // More than 3 chars cannot be root
166  m_path = m_path.substr(0, len - 1);
167  }
168  else { // Only drive root?
169  if (m_path[1] == ':') {
170  // Here m_path must be a drive root. The colon ":" is not allowed in pathes on Windows except as the 2nd char to denote the drive letter
171  if (len == 2) { // Whoa -- "X:" misses leading / for drive root, append it
172  m_path.append("/");
173  }
174  else if (len == 3 && m_path[2] != '/') { // Whoa -- "X:f" misses leading / for root directory, insert it
175  m_path.insert(2, "/");
176  }
177  // else length is 3 with a slash, i.e. "X:/". This is fine and shall not be touched.
178  }
179  else { // not a drive root, delete trailing / if any
180  if (m_path[len - 1] == '/') {
181  m_path = m_path.substr(0, len - 1);
182  }
183  }
184  }
185  }
186 #else
187 #error Unsupported system
188 #endif
189 }

433 std::vector<Path> Path::burst(bool descend /* = false */) const
434 {
435  size_t pos = 0;
436  size_t lastpos = 0;
437  std::vector<Path> results;
438  std::string prefix;
439 
440  // Take care of leading / of absolute paths
441  if (m_path[0] == '/') {
442  results.push_back(Path("/"));
443  prefix.append("/");
444 
445  // Adjust pos so we don’t find the initial /
446  pos++;
447  lastpos++;
448  }
449 
450  while((pos = m_path.find("/", pos)) != string::npos) {
451  std::string component = m_path.substr(lastpos, pos - lastpos);
452 
453  if (descend) {
454  results.push_back(Path(prefix + component));
455  prefix.append(component);
456  prefix.append("/");
457  }
458  else {
459  results.push_back(Path(component));
460  }
461 
462  lastpos = pos + 1;
463  pos++;
464  }
465 
466  std::string lastcomponent = m_path.substr(lastpos);
467 
468  if (descend)
469  results.push_back(Path(prefix + lastcomponent)); // Note no trailing /
470  else
471  results.push_back(Path(lastcomponent));
472 
473  return results;
474 }

507 Path Path::prune() const
508 {
509  std::string newpath(m_path); // copy
510  size_t pos = 0;
511  while((pos = newpath.find("/.", pos)) != string::npos) { // assignment intended
512  if (newpath.substr(pos, 3) == "/..") {
513 
514  // Weird path like /..foo or foo/..bar, which are NOT relative pathes
515  if (newpath.length() > pos + 3 && newpath[pos + 3] != '/') {
516  // Do not reset `pos' -- this has to stay. Advance to the next char.
517  pos++;
518  continue;
519  }
520 
521  if (pos == 0) {
522  // /.. at beginning of string, replace with root / (/ on Windows is root on current drive)
523  newpath.erase(pos, 3);
524 
525  // Whoops -- the entire string was just "/.."
526  if (newpath.empty()) {
527  newpath.append("/");
528  }
529  }
530 #ifdef _WIN32
531  // Cater for pathes with drive X:/ on Windows
532  else if (pos == 2 && newpath[1] == ':') { // ":" is on Windows only allowed at pos 1, where it signifies the preceding char is a drive letter
533  // X:/. or X:/.. at beginning of string
534  if(newpath.length() > 4 && newpath[4] == '.') { // X:/..
535  // Prevent special case "X:/..foo", which is directory "..foo" under the root
536  if (newpath.length() <= 5 || newpath[5] != '/') {
537  // X:/.. or X:/../foo/bar at beginning of string, replace with drive root
538  newpath.erase(pos, 3);
539  }
540  }
541  else { // X:/./foo/bar X:/..foo
542  // Prevent special case "X:/.foo", which is directory ".foo" under the root
543  if (newpath.length() <= 4 || newpath[4] != '/') {
544  // X:/. or X:/./foo/bar at beginning of string, replace with drive root
545  newpath.erase(pos, 2);
546  }
547  }
548 
549  if (newpath.length() == 2) {
550  // Whoops -- the entire string was just "X:/.." or "X:/."
551  newpath.append("/");
552  }
553  }
554 #endif
555  else {
556  size_t pos2 = 0;
557  if ((pos2 = newpath.rfind("/", pos - 1)) != string::npos) { // assignment intended
558  // Remove parent directory.
559  newpath.erase(pos2, pos - pos2 + 3);
560  }
561  else { // ../ for relative path (as in foo/../baz.txt)
562  newpath.erase(0, pos + 4);
563  }
564  }
565  }
566  else { // Single /.
567 
568  // Weird path like /..foo or foo/..bar, which are NOT relative pathes
569  if (newpath.length() > pos + 2 && newpath[pos + 2] != '/') {
570  // Do not reset `pos' -- this has to stay. Advance to the next char.
571  pos++;
572  continue;
573  }
574 
575  newpath.erase(pos, 2);
576 
577  // Whoops -- the entire string was just "/."
578  if (newpath.empty()) {
579  newpath.append("/");
580  }
581  }
582 
583  // Reset as we have modified the string and might need to go again over it
584  pos = 0;
585  }
586 
587  /* If we are empty now, the original string was a one-element
588  * relative path with .. appended. We cannot know what to set
589  * without referring to pwd(), which is external access and
590  * forbidden for this method. So instead, we do the one sane thing
591  * and just use ".". */
592  if (newpath.empty())
593  newpath = ".";
594 
595  return Path(newpath);
596 }

703 Path Path::real() const
704 {
705 #if defined(_PATHIE_UNIX)
706  std::string nstr = native();
707  char path[PATH_MAX];
708  if (!realpath(nstr.c_str(), path))
709  throw(Pathie::ErrnoError(errno));
710 
711  return Path(filename_to_utf8(path));
712 #elif defined(_WIN32)
713  // On Windows there sadly is no easy way to do this. We can
714  // only determine if a given path is a symlink and resolve it...
715  // Instructions taken from: http://msdn.microsoft.com/en-us/library/windows/desktop/aa363940%28v=vs.85%29.aspx
716  std::vector<Path> components = burst();
717  unsigned int pos = 0;
718 
719  while (pos < components.size()) {
720  // Build path consisting of all elements upto our position pointer
721  Path reduced_path(components.front());
722  if (components.size() - pos > 1) {
723  for (unsigned int i=1; i <= pos; i++) { // i=0 is already in the initialization above
724  reduced_path = reduced_path.join(components[i]);
725  }
726  }
727 
728  // If that’s a symlink, resolve it and replace our path until
729  // the symlink with the symlink’s target.
730  std::wstring reduced_path_utf16 = utf8_to_utf16(reduced_path.m_path);
731  if (is_ntfs_symlink(reduced_path_utf16.c_str())) {
732  wchar_t* target_utf16 = read_ntfs_symlink(reduced_path_utf16.c_str());
733  Path target(utf16_to_utf8(target_utf16));
734  std::vector<Path> target_components = target.burst();
735  free(target_utf16);
736 
737  // Replace all components up to pos with the symlink target
738  components.erase(components.begin(), components.begin() + pos);
739  std::vector<Path> temp(components);
740  components.clear();
741  for(auto iter=target_components.begin(); iter != target_components.end(); iter++)
742  components.push_back(*iter);
743  for(auto iter=temp.begin(); iter != temp.end(); iter++)
744  components.push_back(*iter);
745  }
746  else {
747  // Note a symlink can point to another symlink, so we can only
748  // advance to the next element if this element has been tested
749  // for not being a symlink.
750  pos++;
751  }
752  }
753 
754  // BUild a new path from the now resolved components
755  Path result(components.front());
756  if (components.size() > 1) {
757  for(auto iter=components.begin(); iter != components.end(); iter++) {
758  result = result.join(*iter);
759  }
760  }
761 
762  return result;
763 #else
764 #error Unsupported system.
765 #endif
766 }

768 #ifdef __WIN32
769 /*
770  * Checking if a file is a symlink under Windows is insane.
771  * See http://msdn.microsoft.com/en-us/library/windows/desktop/aa363940%28v=vs.85%29.aspx
772  * for the detailed instructions by Microsoft on how to do
773  * that.
774  */
775 bool Path::is_ntfs_symlink(const wchar_t* path) const
776 {
777  // First we need to obtain the file attributes.
778  DWORD attrs = GetFileAttributesW(path);
779  if (attrs == INVALID_FILE_ATTRIBUTES) {
780  DWORD err = GetLastError();
781  throw(Pathie::WindowsError(err));
782  }
783 
784  /* These file attributes must contain the REPARSE_POINT attribute
785  * that mark the file as being symlink, junction, or similar.
786  * Actually, reparse points can contain many more custom data, but
787  * we are not intersted in those. */
788  if (attrs & FILE_ATTRIBUTE_REPARSE_POINT) {
789  // Now we have to retrieve a special attributes handle from the file.
790  WIN32_FIND_DATAW finddata;
791  HANDLE findhandle = FindFirstFileW(path, &finddata);
792  if (findhandle == INVALID_HANDLE_VALUE) {
793  DWORD err = GetLastError();
794  throw(Pathie::WindowsError(err));
795  }
796  FindClose(findhandle);
797 
798  // These extended attributes contain the SYMLINK tag if this file
799  // is a symlink.
800  if (finddata.dwReserved0 & IO_REPARSE_TAG_SYMLINK)
801  return true;
802 
803  // Junction or so, we do not resolve that
804  return false;
805  }
806 
807  // Regular file
808  return false;
809 }

811 /*
812  * Reading the link target also is insanely hard.
813  * The process is documented at http://msdn.microsoft.com/en-us/library/windows/desktop/aa365503%28v=vs.85%29.aspx
814  * in general. The key function is DeviceIoControl(), documented
815  * at http://msdn.microsoft.com/en-us/library/windows/desktop/aa363216%28v=vs.85%29.aspx
816  * .
817  *
818  * This function does not check if `path` is a symlink, but assumes it.
819  * It will exhibit unexpactable behaviour if this assumption is wrong.
820  *
821  * The returned pointer must be freed by you.
822  */
823 wchar_t* Path::read_ntfs_symlink(const wchar_t* path) const
824 {
825  // We have to open the file (directories are files on Windows also) first.
826  HANDLE filehandle = CreateFileW(path, GENERIC_READ, 0, NULL, OPEN_EXISTING, FILE_FLAG_OPEN_REPARSE_POINT, NULL);
827  if (filehandle == INVALID_HANDLE_VALUE) {
828  DWORD err = GetLastError();
829  throw(Pathie::WindowsError(err));
830  }
831 
832  // This infamous structure is documented here: http://msdn.microsoft.com/en-us/library/ff552012.aspx
833  unsigned long reparsebufsize = REPARSE_GUID_DATA_BUFFER_HEADER_SIZE; // According to docs this is the minimum size
834  REPARSE_DATA_BUFFER* p_reparse_data = NULL;
835  while (true) {
836  reparsebufsize += 4096; // Do you have a better guess?
837  p_reparse_data = (REPARSE_DATA_BUFFER*) realloc(p_reparse_data, reparsebufsize);
838  memset(p_reparse_data, '\0', reparsebufsize);
839 
840  DWORD bytecount = 0;
841  // Obtain the reparse tag. FSCTL_GET_REPARSE_POINT is documented here: http://msdn.microsoft.com/en-us/library/windows/desktop/aa364571(v=vs.85).aspx
842  if (DeviceIoControl(filehandle, FSCTL_GET_REPARSE_POINT, NULL, 0, p_reparse_data, reparsebufsize, &bytecount, NULL) == 0) {
843  DWORD errsav = GetLastError();
844  if (errsav == ERROR_INSUFFICIENT_BUFFER) { // buffer was to small, try again
845  continue;
846  }
847  else {
848  throw(Pathie::WindowsError(errsav));
849  }
850  }
851  else { // success
852  break;
853  }
854  }
855 
856  // See also http://msdn.microsoft.com/en-us/library/windows/desktop/aa365511(v=vs.85).aspx
857  // And this one: http://www.codeproject.com/Articles/21202/Reparse-Points-in-Vista
858  if (p_reparse_data->ReparseTag == IO_REPARSE_TAG_SYMLINK) {
859  wchar_t* subsname = (wchar_t*) malloc(p_reparse_data->SymbolicLinkReparseBuffer.SubstituteNameLength + 2); // UTF-16 NUL
860  wchar_t* printname = (wchar_t*) malloc(p_reparse_data->SymbolicLinkReparseBuffer.PrintNameLength + 2); // UTF-16 NUL
861 
862  memset(subsname, '\0', p_reparse_data->SymbolicLinkReparseBuffer.SubstituteNameLength + 2);
863  memset(printname, '\0', p_reparse_data->SymbolicLinkReparseBuffer.PrintNameLength + 2);
864 
865  wcsncpy(subsname, &p_reparse_data->SymbolicLinkReparseBuffer.PathBuffer[p_reparse_data->SymbolicLinkReparseBuffer.SubstituteNameOffset], p_reparse_data->SymbolicLinkReparseBuffer.SubstituteNameLength / sizeof(WCHAR));
866  wcsncpy(printname, &p_reparse_data->SymbolicLinkReparseBuffer.PathBuffer[p_reparse_data->SymbolicLinkReparseBuffer.PrintNameOffset], p_reparse_data->SymbolicLinkReparseBuffer.PrintNameLength / sizeof(WCHAR));
867 
868  // Actually, it appears the subsname has no real usecase...
869  free(subsname);
870  free(p_reparse_data);
871  CloseHandle(filehandle);
872  return printname;
873  }
874  else {
875  return NULL;
876  }
877 }

1051 Path Path::relative(Path base) const
1052 {
1053  if (is_relative())
1054  throw(std::invalid_argument("Referenced path must be absolute."));
1055  if (base.is_relative())
1056  throw(std::invalid_argument("Argument path must be absolute."));
1057 
1058  // Wipe all ".." and ".", this would break the below algorithm
1059  base = base.prune();
1060  Path refpath = prune();
1061 
1062  // Shortcut for equal paths
1063  if (base.m_path == refpath.m_path)
1064  return Path(".");
1065 
1066  // Shortcut if base is the root
1067  if (base.is_root()) {
1068 #if defined(_PATHIE_UNIX)
1069  return Path(refpath.m_path.substr(1)); // Skip leading /
1070 #elif defined(_WIN32)
1071  return Path(refpath.m_path.substr(root().m_path.length())); // Skip leading / or X:/
1072 #else
1073 #error Unsupported system.
1074 #endif
1075  }
1076 
1077  size_t pos = 0;
1078  size_t baselength = base.m_path.length();
1079  size_t reflength = refpath.m_path.length();
1080  while (true) {
1081  if (pos >= baselength)
1082  break;
1083  else if (pos >= reflength)
1084  break;
1085  else if (base.m_path[pos] != refpath.m_path[pos])
1086  break;
1087  else
1088  pos++;
1089  }
1090  // pos now points to the last character in which both strings were equal
1091 
1092  // For each component in base that is not part of refpath, add a "..".
1093  std::string resultstr;
1094  Path basepart(base.m_path.substr(pos));
1095  for(size_t i=0; i < basepart.component_count(); i++)
1096  resultstr.append("../");
1097 
1098  // Now append the part of refpath that is not part of base to the result.
1099  resultstr.append(refpath.m_path.substr(pos));
1100 
1101  // Done.
1102  return Path(resultstr);
1103 }
