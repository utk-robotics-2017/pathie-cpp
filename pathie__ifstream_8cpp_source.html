227  void Pathie::ifstream::open(const char* filename, ios_base::openmode mode)
228  {
229  std::wstring w_filename = Pathie::utf8_to_utf16(filename);
230 
231  mp_file = _wfopen(w_filename.c_str(), L"r"); // Mode will be overridden
232  if (!mp_file) {
233  setstate(ios_base::failbit);
234  return;
235  }
236 
237  /* The following construction uses a “placement new” as it appears
238  * to be the only "clean" solution applicable. The init() method,
239  * an internum of the GCC implementation of basic_istream that
240  * needs to be called in the stream’s constructor, requires a
241  * pointer to the filebuffer object. However, we do not have that
242  * filebuffer object at hand in the constructor, the
243  * __gnu_cxx::stdio_filebuf instance will be created later when
244  * open() is called. It is impossible to construct it earlier,
245  * because it does not support a delayed open() call, the file
246  * descriptor or FILE* pointer must be passed during its
247  * construction, but we don’t have it there; it is available only
248  * in open() -- remember that you can create the ifstream instance
249  * without being attached to a file and then call open() later
250  * with a filename. To be able to pass something meaningful to
251  * init(), we have to "foresee" where in memory the stdio_filebuf
252  * instance will be created. This only is possible with a
253  * placement new into a place we have allocated previously using
254  * malloc().
255  *
256  * An alternative would be to use internal GCC APIs by duplicating
257  * the sourcecode of the __gnu_cxx::stdio_filebuf constructor; however
258  * undocumented internal APIs are never good to use. For informational
259  * purposes therefore the sourcecode link:
260  *
261  * https://gcc.gnu.org/onlinedocs/gcc-4.9.2/libstdc++/api/a01222_source.html
262  */
263 
264  new (mp_filebuffer) __gnu_cxx::stdio_filebuf<char>(mp_file, mode);
265  m_buffer_allocated = true;
266 
267  if (!mp_filebuffer->is_open())
268  setstate(ios_base::failbit);
269  else
270  clear();
271  }
